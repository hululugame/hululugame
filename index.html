const canvas=document.getElementById("bg") const ctx=canvas.getContext("2d") canvas.width=window.innerWidth canvas.height=window.innerHeight let stars=[] let fireworks=[] for(let i=0;i<100;i++){ stars.push({ x:Math.random()*canvas.width, y:Math.random()*canvas.height, r:Math.random()*2 }) } function createFirework(x,y){ for(let i=0;i<40;i++){ fireworks.push({ x:x, y:y, dx:(Math.random()-0.5)*6, dy:(Math.random()-0.5)*6, life:80, color:`hsl(${Math.random()*360},100%,60%)` }) } } canvas.addEventListener("click",e=>{ createFirework(e.clientX,e.clientY) }) setInterval(()=>{ createFirework( Math.random()*canvas.width, Math.random()*canvas.height/2 ) },1500) function animate(){ ctx.clearRect(0,0,canvas.width,canvas.height) /* 星空 */ ctx.fillStyle="white" stars.forEach(s=>{ ctx.beginPath() ctx.arc(s.x,s.y,s.r,0,Math.PI*2) ctx.fill() }) /* 煙火 */ fireworks.forEach((p,i)=>{ p.x+=p.dx p.y+=p.dy p.life-- ctx.beginPath() ctx.arc(p.x,p.y,2,0,Math.PI*2) ctx.fillStyle=p.color ctx.fill() if(p.life<=0){ fireworks.splice(i,1) } }) requestAnimationFrame(animate) } animate()
